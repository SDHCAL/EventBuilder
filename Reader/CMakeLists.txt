# project name
PROJECT( Reader )

# project version

file(COPY ./include/ DESTINATION ${PROJECT_INCLUDE_DIR})

TARGET_LINK_LIBRARIES(Reader Geometry Tinyxml )
INSTALL(TARGETS Reader DESTINATION ${LIBRARY_OUTPUT_PATH} )
